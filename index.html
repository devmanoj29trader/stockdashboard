        // Firebase Imports

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getFirestore, doc, collection, onSnapshot, addDoc, deleteDoc, setLogLevel } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // --- GLOBAL VARIABLES ---

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : { apiKey: "DEMO", authDomain: "DEMO", projectId: "DEMO" };

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'stock-alert-dashboard';

        

        let db, auth, userId, tickersCollectionRef;

        let apiKey = localStorage.getItem('alphaVantageApiKey') || '';

        const tickersData = new Map();


        // --- DOM ELEMENTS ---

        const apiKeySection = document.getElementById('api-key-section');

        const apiKeyInput = document.getElementById('api-key-input');

        const saveApiKeyBtn = document.getElementById('save-api-key-btn');

        const dashboardContent = document.getElementById('dashboard-content');

        const tickerInput = document.getElementById('ticker-input');

        const addTickerBtn = document.getElementById('add-ticker-btn');

        const tickerList = document.getElementById('ticker-list');

        const addTickerError = document.getElementById('add-ticker-error');

        const refreshAllBtn = document.getElementById('refresh-all-btn');

        const noTickersMessage = document.getElementById('no-tickers-message');

        const alertModal = document.getElementById('alert-modal');

        const modalBody = document.getElementById('modal-body');

        const closeModalBtn = document.getElementById('close-modal-btn');

        const userIdDisplay = document.getElementById('user-id-display');


        // --- INITIALIZATION ---

        async function initialize() {

            // Initialize Firebase

            const app = initializeApp(firebaseConfig);

            db = getFirestore(app);

            auth = getAuth(app);

            setLogLevel('debug');


            // Handle Authentication

            onAuthStateChanged(auth, async (user) => {

                if (user) {

                    userId = user.uid;

                    console.log("User is signed in with UID:", userId);

                    tickersCollectionRef = collection(db, `artifacts/${appId}/users/${userId}/tickers`);

                    userIdDisplay.textContent = `User ID: ${userId}`;

                    setupListeners();

                    checkApiKey();

                } else {

                    console.log("No user signed in, attempting anonymous sign-in.");

                     try {

                        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                            await signInWithCustomToken(auth, __initial_auth_token);

                        } else {

                            await signInAnonymously(auth);

                        }

                    } catch (error) {

                        console.error("Error during sign-in:", error);

                        // Display an error to the user that authentication failed.

                    }

                }

            });

        }


        function checkApiKey() {

            if (apiKey) {

                apiKeySection.classList.add('hidden');

                dashboardContent.classList.remove('hidden');

            } else {

                apiKeySection.classList.remove('hidden');

                dashboardContent.classList.add('hidden');

            }

        }


        // --- EVENT LISTENERS ---

        function setupListeners() {

            saveApiKeyBtn.addEventListener('click', () => {

                const key = apiKeyInput.value.trim();

                if (key) {

                    apiKey = key;

                    localStorage.setItem('alphaVantageApiKey', key);

                    checkApiKey();

                }

            });


            addTickerBtn.addEventListener('click', handleAddTicker);

            tickerInput.addEventListener('keyup', (e) => {

                if (e.key === 'Enter') handleAddTicker();

            });


            refreshAllBtn.addEventListener('click', refreshAllData);

            closeModalBtn.addEventListener('click', () => alertModal.classList.add('hidden'));


            // Firestore listener

            onSnapshot(tickersCollectionRef, (snapshot) => {

                const remoteTickers = new Set();

                snapshot.docs.forEach(doc => {

                    const ticker = doc.data().symbol.toUpperCase();

                    remoteTickers.add(ticker);

                    if (!tickersData.has(ticker)) {

                        tickersData.set(ticker, { id: doc.id, symbol: ticker, status: 'loading' });

                        fetchStockData(ticker);

                    }

                });

                

                // Remove local tickers that are no longer in Firestore

                for (const localTicker of tickersData.keys()) {

                    if (!remoteTickers.has(localTicker)) {

                        tickersData.delete(localTicker);

                    }

                }


                renderTickerList();

            });

        }


        // --- CORE LOGIC ---

        async function handleAddTicker() {

            const symbol = tickerInput.value.trim().toUpperCase();

            if (!symbol) return;

            

            addTickerError.textContent = '';

            if (tickersData.has(symbol)) {

                addTickerError.textContent = `${symbol} is already being monitored.`;

                return;

            }


            addTickerBtn.disabled = true;

            addTickerBtn.innerHTML = '<div class="loader mx-auto"></div>';


            try {

                // First, validate ticker with API

                const quote = await fetchQuote(symbol);

                if (!quote || !quote['Global Quote'] || !quote['Global Quote']['01. symbol']) {

                    throw new Error(`Invalid ticker symbol or API error for ${symbol}.`);

                }


                // If valid, add to Firestore

                await addDoc(tickersCollectionRef, { symbol: symbol, createdAt: new Date() });

                tickerInput.value = '';


            } catch (error) {

                console.error("Error adding ticker:", error);

                addTickerError.textContent = error.message;

            } finally {

                addTickerBtn.disabled = false;

                addTickerBtn.textContent = 'Add Ticker';

            }

        }


        async function handleRemoveTicker(symbol) {

            const ticker = tickersData.get(symbol);

            if (ticker && ticker.id) {

                try {

                    await deleteDoc(doc(db, `artifacts/${appId}/users/${userId}/tickers`, ticker.id));

                    // The onSnapshot listener will handle the UI update

                } catch (error) {

                    console.error("Error removing ticker:", error);

                }

            }

        }


        async function fetchWithBackoff(url, retries = 3, delay = 1000) {

            for (let i = 0; i < retries; i++) {

                try {

                    const response = await fetch(url);

                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

                    const data = await response.json();

                     // Alpha Vantage returns an error message in the response body for invalid calls

                    if (data["Error Message"] || data["Information"]) {

                        console.warn(`API Info/Error for ${url}:`, data["Error Message"] || data["Information"]);

                        // For API limit messages, we should retry

                        if (data["Information"] && data["Information"].includes('API call frequency')) {

                           await new Promise(res => setTimeout(res, delay * (i + 1)));

                           continue;

                        }

                        return null; // Don't retry for other errors like invalid symbol

                    }

                    return data;

                } catch (error) {

                    console.error(`Fetch attempt ${i + 1} failed:`, error);

                    if (i < retries - 1) {

                        await new Promise(res => setTimeout(res, delay * (i + 1)));

                    } else {

                        throw error;

                    }

                }

            }

            // If all retries fail, return null

            return null;

        }

        

        async function fetchQuote(symbol) {

            const url = `https://www.alphavantage.co/query?function=GLOBAL_QUOTE&symbol=${symbol}&apikey=${apiKey}`;

            return await fetchWithBackoff(url);

        }


        async function fetchTimeSeries(symbol) {

            const url = `https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=${symbol}&apikey=${apiKey}`;

            return await fetchWithBackoff(url);

        }


        async function fetchStockData(symbol) {

            tickersData.set(symbol, { ...tickersData.get(symbol), status: 'loading' });

            renderTickerList();

            

            try {

                const [quoteData, timeSeriesData] = await Promise.all([

                    fetchQuote(symbol),

                    fetchTimeSeries(symbol)

                ]);


                if (!quoteData || !quoteData['Global Quote'] || Object.keys(quoteData['Global Quote']).length === 0) {

                     throw new Error('Could not fetch current price. The API may have reached its call limit.');

                }

                if (!timeSeriesData || !timeSeriesData['Time Series (Daily)']) {

                     throw new Error('Could not fetch historical data. The API may have reached its call limit.');

                }


                const quote = quoteData['Global Quote'];

                const currentPrice = parseFloat(quote['05. price']);

                

                const dailyData = timeSeriesData['Time Series (Daily)'];

                const dates = Object.keys(dailyData).slice(0, 30);

                if(dates.length < 30) {

                    console.warn(`Not enough data for a 30-day average for ${symbol}. Using ${dates.length} days.`);

                }

                const sum = dates.reduce((acc, date) => acc + parseFloat(dailyData[date]['4. close']), 0);

                const sma30 = sum / dates.length;


                const diff = ((currentPrice - sma30) / sma30) * 100;

                const isAlert = currentPrice <= sma30 * 0.8;


                tickersData.set(symbol, { 

                    ...tickersData.get(symbol), 

                    status: 'loaded', 

                    currentPrice, 

                    sma30, 

                    diff, 

                    isAlert 

                });


                if (isAlert) {

                    showModalAlert(symbol, currentPrice, sma30);

                }


            } catch (error) {

                console.error(`Failed to fetch data for ${symbol}:`, error);

                tickersData.set(symbol, { ...tickersData.get(symbol), status: 'error', errorMessage: error.message });

            }

            renderTickerList();

        }


        function refreshAllData() {

            refreshAllBtn.disabled = true;

            refreshAllBtn.querySelector('svg').classList.add('animate-spin');

            

            const fetchPromises = [];

            for (const symbol of tickersData.keys()) {

                fetchPromises.push(fetchStockData(symbol));

            }


            Promise.all(fetchPromises).finally(() => {

                refreshAllBtn.disabled = false;

                refreshAllBtn.querySelector('svg').classList.remove('animate-spin');

            });

        }


        // --- UI RENDERING ---

        function renderTickerList() {

            tickerList.innerHTML = '';

            if (tickersData.size === 0) {

                noTickersMessage.classList.remove('hidden');

                return;

            }

            noTickersMessage.classList.add('hidden');


            const sortedTickers = [...tickersData.entries()].sort();


            for (const [symbol, data] of sortedTickers) {

                const card = document.createElement('div');

                card.className = 'card flex flex-col justify-between';

                card.id = `ticker-${symbol}`;


                let content = '';

                if (data.status === 'loading') {

                    content = `

                        <div class="flex justify-between items-center mb-4">

                            <h3 class="text-2xl font-bold text-white">${symbol}</h3>

                            <div class="loader"></div>

                        </div>

                        <p class="text-gray-400">Fetching data...</p>

                    `;

                } else if (data.status === 'error') {

                    content = `

                        <div class="flex justify-between items-center mb-4">

                            <h3 class="text-2xl font-bold text-white">${symbol}</h3>

                            <span class="text-red-400 font-bold">Error</span>

                        </div>
